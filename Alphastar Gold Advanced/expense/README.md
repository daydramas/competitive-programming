		int mid = (min+max)/2;

		int sum = 0;
		int group = 1;

			if(spend[i]>mid) {

				group++, sum = 0;
			}
			sum += spend[i];
		}
		if(group <= M) {
			max = mid;
		} else {
			min = mid+1;
		}
	}

	// output
	cout << min << endl;
	return 0;
}
