#include <bits/stdc++.h>
using namespace std;

// structures/defines

// global variables
int N, c[100000];
int step[100000];
//functions
int getDist(int i, int j) {
	if (j > i)
		return j - i;
	else
		return N - i + j;
}
int main() {
	// local variables
	queue<int> Q;

	// fstream
	ifstream fin("cbarn.in");
	ofstream fout("cbarn.out");

	// input
	fin >> N;
	for (int i = 0; i < N; i++) {
		fin >> c[i];
		step[i] = 0;
	}
	for (int i = N - 1; i >= 0; i--) {
		if (c[i] == 0) {
			Q.push(i);
		}
	}
	// main
	long long ans = 0;
	while (!Q.empty()) {
		int t = Q.front();
		Q.pop();
		for (int i = (t - 1 + N) % N; i != t; i = (i - 1 + N) % N) {
			if (c[i]) {
				c[i]--, c[t]++;
				step[t] = step[i] + getDist(i, t);
				ans += step[t] * step[t] - step[i] * step[i];
				if (c[i] == 0) {
					Q.push(i);
				}
				break;
			}
		}
	}

	// output
	fout << ans << endl;
	return 0;
}

	int len, xy, xy2;
	edge(int len, int xy, int xy2) :
			len(len), xy(xy), xy2(xy2) {
	}
};
struct s2b {
	bool operator()(edge i, edge j) {
		return (i.len < j.len);
	}
} s2b;
// global variables

int a[2010], b[2010];
pii pa[2010][2010];
//int pa_n[2010][2010];

pii findPa(int i, int j) {
	if (pa[i][j] == pii(i, j)) {
		return pa[i][j];
	} else {
		pa[i][j] = findPa(pa[i][j].first, pa[i][j].second);
		return pa[i][j];
	}
}
void merge(pii a, pii b) {
	pii f1 = findPa(b.first, b.second);
	pa[f1.first][f1.second] = findPa(a.first, a.second);
}
