int n;

void dfs(int sc, int pa) {
    s[sc] = 1;

        dfs(ch, sc);
        s[sc] += s[ch];
        vs[sc].pb(s[ch]);
    }
    if (s[sc]!=n) vs[sc].pb(n-s[sc]);
}
bool ok(int k) {
    if ((n-1)%k) return 0;
    int c[100005]; memset(c, 0, sizeof(c));

        int cnt = 0;
        for(int d : vs[sc]) {
            int m = d%k; if (!m) continue;
            if (c[k-m]) { c[k-m]--; cnt--; }
            else { c[m]++, cnt++; }
        }
        if (cnt!=0) return 0;
    }
    return 1;
}
