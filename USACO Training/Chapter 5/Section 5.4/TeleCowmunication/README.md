#define INF 1000000000

int n, m, c1, c2, t;
int adj[205][205];
int bu[205][205];
int depth[205];

bool con() {
    memset(depth, 0, sizeof(depth));
    queue<int> Q;
    Q.push(0);
    depth[0] = 1;
    while(!Q.empty()) {
        int f=Q.front(); Q.pop();
        for1(c, t) {
            if (adj[f][c] && !depth[c]) {
                depth[c] = depth[f]+1;
                Q.push(c);
            }
        }
    }
    return (depth[t]);
}

int flow(int cur, int mx) {
    if(cur==t) return mx;

    for1(i, t) if (adj[cur][i] && depth[i]==depth[cur]+1) {
        int nf = flow(i, min(adj[cur][i], mx));
        if (nf) {
            adj[cur][i] -= nf;
            adj[i][cur] += nf;
            return nf;
        }
    }
    return 0;
}

int dinic() {
    int ta = 0;
    while(con()) ta += flow(0, INF);
    return ta;
}
