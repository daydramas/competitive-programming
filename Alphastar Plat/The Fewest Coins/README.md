    for1(i,t+1000) dp1[i]=dp2[i]=INF;

        ans = min(ans, dp1[i-t]+dp2[i]);

    if (ans==INF) ans=-1;
    cwl(ans);
    // cout <<"+------ Ending | Time taken: "<<((float)(clock()-startT)/CLOCKS_PER_SEC)<<" seconds \n\n";
    return 0;
}
